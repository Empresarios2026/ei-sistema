Empresarios Inmobiliarios Configuración inicial

Paso 1: Crear la tabla

Ve a supabase.com → Tu proyecto → SQL Editor → pega esto y ejecuta:

create table propiedades ( id uuid default gen_random_uuid() primary key, codigo text, operacion text, tipo text, titulo text, ubicacion text, precio numeric, moneda text default 'USD', area numeric, dormitorios int, banos int, estacionamiento int, agente text, descripcion text, foto_url text, activo boolean default true, created_at timestamptz default now() ); alter table propiedades enable row level security; create policy "read_all" on propiedades for select using (true); create policy "insert_all" on propiedades for insert with check (true); create policy "update_all" on propiedades for update using (true); create policy "delete_all" on propiedades for delete using (true);
